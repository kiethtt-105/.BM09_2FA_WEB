""" views_otp_history.py — Snippet view admin_otp_history đã cập nhật =================================================================== Copy function này vào views.py, thay thế hàm admin_otp_history cũ. Thay đổi so với phiên bản cũ: - Không truyền full_secret vào context (tránh lộ plaintext vô tình). Template nhận raw_secret từ decrypt_secret() để hiển thị TOTP realtime client-side. - Không đọc otp_code (luôn rỗng trong DB sau khi model.save() xóa). - google_auths dùng key 'raw_secret' thay vì 'full_secret' để rõ ràng hơn. """ from django.contrib.auth.decorators import login_required, user_passes_test from django.core.paginator import Paginator from django.utils import timezone from datetime import timedelta from .models import EmailOTP, UserProfile from .utils import get_totp_token # alias của compute_totp @login_required @user_passes_test(lambda u: u.is_superuser) def admin_otp_history(request): """ Lịch sử OTP với bộ lọc username, trạng thái, ngày. Bảo mật: - otp_code không được đọc (luôn rỗng trong DB). - Chỉ đọc otp_hash để hiển thị. - raw_secret truyền vào template để JS tính TOTP realtime (decrypt phía server). - encrypted_secret truyền vào để show giá trị thực trong DB. """ email_otp_queryset = EmailOTP.objects.select_related('user').order_by('-created_at') username_query = request.GET.get('username', '').strip() status_query = request.GET.get('status', '') date_query = request.GET.get('date', '') if username_query: email_otp_queryset = email_otp_queryset.filter( user__username__icontains=username_query ) if status_query == 'used': email_otp_queryset = email_otp_queryset.filter(is_used=True) elif status_query == 'pending': email_otp_queryset = email_otp_queryset.filter( is_used = False, is_active = True, created_at__gt = timezone.now() - timedelta(minutes=EmailOTP.OTP_VALID_MINUTES) ) elif status_query == 'expired': email_otp_queryset = email_otp_queryset.filter( is_used=False, created_at__lt=timezone.now() - timedelta(minutes=EmailOTP.OTP_VALID_MINUTES) ) if date_query: email_otp_queryset = email_otp_queryset.filter(created_at__date=date_query) paginator = Paginator(email_otp_queryset, 15) email_otps = paginator.get_page(request.GET.get('page')) # Google Authenticator section # Truyền raw_secret (đã decrypt) để JS tính TOTP realtime client-side. # Không có cột otp_code — tất cả đã bị xóa sau khi hash. google_auths = [] for profile in UserProfile.objects.filter(has_app_otp=True).select_related('user'): raw_secret = profile.decrypt_secret() # decrypt Fernet → Base32 plaintext encrypted_secret = profile.otp_secret or '' # Fernet ciphertext (giá trị lưu DB) if raw_secret: masked = raw_secret[:4] + '****' + raw_secret[-4:] else: masked = '— không có secret —' google_auths.append({ 'username': profile.user.username, 'masked_secret': masked, 'raw_secret': raw_secret or '', # để JS tính TOTP realtime 'encrypted_secret': encrypted_secret, # Fernet — giá trị thực trong DB 'current_totp': get_totp_token(raw_secret) if raw_secret else '——', }) context = { 'email_otps': email_otps, 'google_auths': google_auths, 'total_otp': email_otp_queryset.count() + len(google_auths), 'paginator': paginator, 'page_obj': email_otps, } return render(request, 'admin_dashboard/otp_history.html', context)
